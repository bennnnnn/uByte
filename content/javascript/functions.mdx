## Function Declaration

```javascript
function add(a, b) {
  return a + b;
}
console.log(add(3, 4)); // 7
```

## Arrow Functions

```javascript
const multiply = (a, b) => a * b;
const square = n => n * n;
```

## Default Parameters

```javascript
function greet(name = "World") {
  console.log(`Hello, ${name}!`);
}
```

## Closures

```javascript
function counter() {
  let count = 0;
  return () => ++count;
}
const inc = counter();
console.log(inc()); // 1
console.log(inc()); // 2
```
