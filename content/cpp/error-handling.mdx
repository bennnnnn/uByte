## try / catch / throw

```cpp
#include <stdexcept>

void divide(int a, int b) {
    if (b == 0) throw std::invalid_argument("division by zero");
    std::cout << a / b << std::endl;
}

int main() {
    try {
        divide(10, 0);
    } catch (const std::exception& e) {
        std::cout << "Error: " << e.what() << std::endl;
    }
}
```
